Extract complex migration workflow into components that

  • Launch on service startup
  • Fetch all tenants and for each:
  • Setup context for each tenant
    • Multitenancy filter & logging
    • One transaction per tenant
    • Skip and log problematic tenants
  • Fetch all data to be migrated and for each:
  • Apply migration logic to data

class ProductPriceMigrator extends EntityMigrator<Product> {
    Iterable<Product> findEntitiesToMigrate() {
        return repository.findByLastModifiedAtBefore(DATE);
    }

    EntityMigrationResult migrateEntity(TransactionStatus tx,
                                        Product product) {
        priceCalculatingEventHandler.doCalculate(product);
        repository.save(product);
        messagePublisher.publishProductUpdatedEvent(product);
        return success(product);
    }
}
              

class CategoryLabelMigrator extends EntityMigrator<Category> {
    Iterable<Category> findEntitiesToMigrate() {
        return repository.findByLabelIsNullAndLastModifiedAtBefore(DATE);
    }

    EntityMigrationResult migrateEntity(TransactionStatus tx,
                                        Category category) {
        String previousLabel = category.getLabel();
        service.normalizeLabel(category);
        repository.save(category);
        messagePublisher.publishUpdatedEvent(category);
        String change = String.format(
          "Updating label for category '%s' from '%s' to '%s'",
          category.getName(), previousLabel, category.getLabel());
        return success(category, change);
    }
}
