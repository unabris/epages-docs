Consuming the Library

def create
  provision_id = BeyondApi::Provisioning.provision(shop_params)
  redirect_to provision_status_path(provision_id)
end
def shop_params
  params.require(:shop)
        .permit(:username, :email, :password, :first_name,
                :last_name, :street, :city, :postcode, :country,
                :locale, :shop_name, :hostname, :custom_hostname,
                :currency, :customer_group)
end
