module Beyond
  module Api
    class Reseller
      def self.generate_credentials(reseller_code)
        response = Beyond::Request.new("/api/reseller-clients",
                                      "post",
                                      reseller_names: [reseller_code]).perform
        Beyond::Credentials.new(JSON.parse(response.body))
      end
    end
  end
end
