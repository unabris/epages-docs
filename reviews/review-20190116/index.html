Use new reseller based endpoints

Before

class Shop
  def self.find(shop_id, options = {})
    response = Beyond::Request.new("/api" +
                                   "/shops/#{shop_id}", "get", options).perform
    Beyond::Shop.new(JSON.parse(response.body))
  end
end

Now

class Shop
  def self.find(reseller_code, shop_id, options = {})
    response = Beyond::Request.new("/api" +
                                   "/resellers/#{reseller_code}" +
                                   "/shops/#{shop_id}", "get", options).perform
    Beyond::Shop.new(JSON.parse(response.body))
  end
end

Update package list endpoint

Why

  • On Krypton package management is country filtered https://...herokuapp.com/packages?country=DE
  • Beyond API was sending a list of countries with all packages nested
  • Krypton needed to do the country filter in the backend

Before

class Package
  def self.all(reseller_code, country_code)
    response = Beyond::Request.new("/api" +
                                   "/resellers/#{reseller_code}" +
                                   "/countries", "get").perform
    Beyond::Packages.new(JSON.parse(response.body), reseller_code, country_code)
  end
end

Now

class Package
  def self.all(reseller_code, country_code)
    response = Beyond::Request.new("/api" +
                                   "/resellers/#{reseller_code}" +
                                   "/countries/#{country_code}" +
                                   "/packages", "get").perform
    Beyond::Packages.new(JSON.parse(response.body))
  end
end

Use Beyond API countries and locales

What

  • We created a Ruby script that calls Beyond API for getting the supported countries
  • The retrieved values are saved into our database
  • The script is automatically executed every day so we can offer up to date values to our resellers

API call

class System
  def self.localization
    response = Beyond::Request.new("/api/supported-countries", "get").perform
    Beyond::Localization.new(JSON.parse(response.body))
  end
end

Response

{
  "supportedCountries" : [ "AT", "DE", "GB", "IE" ],
  "supportedLocales" : [ "de-AT", "de-DE", "en-GB", "en-IE" ]
}
rake update_system_localization
